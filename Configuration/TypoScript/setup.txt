# Use config.tx_extbase to load before ad_google_maps settings.
config.tx_extbase {
	persistence {
		classes {
			# Using mixed-ins hack from Franz Koch to make map table extendable.
			# @see http://lists.typo3.org/pipermail/typo3-project-typo3v4mvc/2010-September/006497.html
			Tx_AdGoogleMaps_Domain_Model_Map {
				mapping.recordType = 0
				subclasses.0 = Tx_AdGoogleMapsPluginPoly_Domain_Model_Map
			}

			Tx_AdGoogleMapsPluginPoly_Domain_Model_Map {
				mapping {
					tableName = tx_adgooglemaps_domain_model_map
					recordType = 0
					columns {
						tx_adgooglemapspluginpoly_info_window_placing_type.mapOnProperty = infoWindowPlacingType
						tx_adgooglemapspluginpoly_info_window_position.mapOnProperty = infoWindowPosition
					}
				}
			}

			# Set type extensions.
			Tx_AdGoogleMaps_Domain_Model_Layer {
				subclasses {
					Tx_AdGoogleMapsPluginPoly_PluginAdapter_LayerBuilder_Polyline = Tx_AdGoogleMapsPluginPoly_Domain_Model_Layer_Polyline
					Tx_AdGoogleMapsPluginPoly_PluginAdapter_LayerBuilder_Polygon = Tx_AdGoogleMapsPluginPoly_Domain_Model_Layer_Polygon
				}
			}

			Tx_AdGoogleMapsPluginPoly_Domain_Model_Layer_Polyline {
				mapping {
					tableName = tx_adgooglemaps_domain_model_layer
					recordType = Tx_AdGoogleMapsPluginPoly_PluginAdapter_LayerBuilder_Polyline
					columns {
						tx_adgooglemapspluginpoly_clickable.mapOnProperty = clickable
						tx_adgooglemapspluginpoly_geodesic.mapOnProperty = geodesic
						tx_adgooglemapspluginpoly_zindex.mapOnProperty = zindex
						tx_adgooglemapspluginpoly_add_markers.mapOnProperty = addMarkers
						tx_adgooglemapspluginpoly_force_listing.mapOnProperty = forceListing
						tx_adgooglemapspluginpoly_info_window_placing_type.mapOnProperty = infoWindowPlacingType
						tx_adgooglemapspluginpoly_info_window_position.mapOnProperty = infoWindowPosition
						tx_adgooglemapspluginpoly_stroke_color.mapOnProperty = strokeColor
						tx_adgooglemapspluginpoly_stroke_opacity.mapOnProperty = strokeOpacity
						tx_adgooglemapspluginpoly_stroke_weight.mapOnProperty = strokeWeight
						tx_adgooglemapspluginpoly_info_window_placing_type.mapOnProperty = infoWindowPlacingType
						tx_adgooglemapspluginpoly_info_window_position.mapOnProperty = infoWindowPosition
					}
				}
			}

			Tx_AdGoogleMapsPluginPoly_Domain_Model_Layer_Polygon < .Tx_AdGoogleMapsPluginPoly_Domain_Model_Layer_Polyline
			Tx_AdGoogleMapsPluginPoly_Domain_Model_Layer_Polygon {
				mapping {
					tableName = tx_adgooglemaps_domain_model_layer
					recordType = Tx_AdGoogleMapsPluginPoly_PluginAdapter_LayerBuilder_Polygon
					columns {
						tx_adgooglemapspluginpoly_fill_color.mapOnProperty = fillColor
						tx_adgooglemapspluginpoly_fill_opacity.mapOnProperty = fillOpacity
					}
				}
			}
		}
	}
}

		plugin {

			# Use includeCSS, cssInline, includeJSlibs, includeJSFooterlibs, includeJS, includeJSFooter, jsInline or jsFooterInline.

			includeFrontEndResources {

				# Use classname where they loaded as unique configuration key.
				Tx_AdGoogleMapsPluginPoly_PluginAdapter_LayerBuilder_Polyline {
					includeJS {

				}

				Tx_AdGoogleMapsPluginPoly_PluginAdapter_LayerBuilder_Polygon < .Tx_AdGoogleMapsPluginPoly_PluginAdapter_LayerBuilder_Polyline

		}
